#include <stdio.h>

int main() {
    int N, i, j, count = 0;
    printf("Enter a number: ");
    scanf("%d", &N);
    for (i = 2; i <= N; i++) {
        int isPrime = 1;
        for (j = 2; j < i; j++) {
            if (i % j == 0) {
                isPrime = 0; 
                break;
            }
        }
        if (isPrime == 1) {
            count++;
        }
    }
    printf("Count of prime numbers between 1 and %d is: %d\n", N, count);
    return 0;
}
