#include <stdio.h>
int main() {
    int rows = 10;
    int i, j;
    for (i = 1; i <= rows; i++) 
    {
        if (i <= 5) 
        {
            for (j = 1; j <= 6 - i; j++) 
            {
                printf("*");
            }
            for (j = 1; j <= 2 * i - 2; j++)
            {
                printf(" ");
            }
            for (j = 1; j <= 6 - i; j++) 
            {
                printf("*");
            }
        }
        else {
            for (j = 1; j <= i - 5; j++) 
            {
                printf("*");
            }
            for (j = 1; j <= 2 * (10 - i) ; j++)
            {
                printf(" ");
            }
            for (j = 1; j <= i - 5; j++) 
            {
                printf("*");
            }
        }
        printf("\n");
    }
    return 0;
}
