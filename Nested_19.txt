#include <stdio.h>
int main() {
    int i, j, spaces;
    int n = 5; 
    for (i = 1; i < n; i++)
    {
        for (j = 0; j < i; j++) printf("*");
        for (spaces = 0; spaces < 2 * (n - i) + 2; spaces++) printf(" ");
        for (j = 0; j < i; j++) printf("*");
        printf("\n");
    }
    for (i = 0; i < 2 * n - 1; i++) printf("*");
    printf("\n");
    for (i = n - 1; i >= 1; i--) {
        for (j = 0; j < i; j++) printf("*");
        for (spaces = 0; spaces < 2 * (n - i) + 2; spaces++) printf(" ");
        for (j = 0; j < i; j++) printf("*");
        printf("\n");
    }
    return 0;
}
