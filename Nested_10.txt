#include <stdio.h>
int main() {
    int rows = 5; 
    int i, j;
    char startChar;
    for (i = 0; i < rows; i++)
    {
        startChar = 'A' + (rows - 1) - i; 
        for (j = 0; j <= i; j++) 
        {
            printf("%c ", startChar + j);
        }
        printf("\n"); 
    }
    return 0;
}
