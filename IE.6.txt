#include <stdio.h>
int main() {
    char c;
    printf("Enter a character: ");
    scanf("%c", &c);

    if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u' ) {
        printf("It is an Vowel,Lower case");
    } 
    else if (c == 'A' || c == 'E' || c == 'I' || c == 'O' || c == 'U')
    {
        printf("It is an Vowel,Upper case");
    }
    else {
        printf("It is consonant");
    }

    return 0;
}
