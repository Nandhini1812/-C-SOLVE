#include <stdio.h>
int main() {
    int n = 4;
    int spaces = 2 * (n - 1);
    for (int i = 1; i <= n; i++)
    {
        for (int j = 1; j <= i; j++)
        {
            printf("%d", j);
        }
        for (int j = 0; j < spaces; j++) 
        {
            printf(" ");
        }
        for (int j = i; j >= 1; j--)
        {
            if (j != n) 
            {
                printf("%d", j);
            }
        }
        printf("\n");
        spaces -= 2;
    }
    return 0;
}
