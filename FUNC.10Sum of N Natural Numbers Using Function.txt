#include <stdio.h>

// Function to calculate sum of first N natural numbers
int sumNatural(int n) {
    int sum = 0;
    for (int i = 1; i <= n; i++) {
        sum += i;
    }
    return sum;
}

int main() {
    int N;
    scanf("%d", &N);
    printf("%d", sumNatural(N));
    return 0;
}
