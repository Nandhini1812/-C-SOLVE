#include <stdio.h>
int main() {
    int n = 7; 
    int i, j, min_dist;
    for (i = 0; i < n; i++) {
        for (j = 0; j < n; j++) {
            int dist_vertical = (i < n - 1 - i) ? i : n - 1 - i;
            int dist_horizontal = (j < n - 1 - j) ? j : n - 1 - j;
            min_dist = (dist_vertical < dist_horizontal) ? dist_vertical : dist_horizontal;
            printf("%d ", 4 - min_dist);
        }
        printf("\n");
    }
    return 0;
}
