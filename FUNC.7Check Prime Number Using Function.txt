#include <stdio.h>

// Function to check prime
void checkPrime(int n) {
    int flag = 1;
    if (n <= 1) flag = 0;
    for (int i = 2; i <= n/2; i++) {
        if (n % i == 0) {
            flag = 0;
            break;
        }
    }
    if (flag == 1)
        printf("Prime");
    else
        printf("Not Prime");
}

int main() {
    int N;
    scanf("%d", &N);
    checkPrime(N);
    return 0;
}
